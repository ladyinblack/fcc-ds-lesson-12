### Solution 1

```js
let myNestedArray2 = [
  'level 1'                       /* myNestedArray2[0] */,
  ['level 2']                     /* myNestedArray2[1][0] */,
  [['level 3', 'deep']]           /* myNestedArray2[2][0][0] */,
  [[['level 4', 'deeper']]]       /* myNestedArray2[3][0][0][0] */,
  [[[['level 5', 'deepest']]]]    /* myNestedArray2[4][0][0][0][0] */
];
```
